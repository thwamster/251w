		\temp
		\begin{proposition}
			Use the definition of a limit to prove $\limn \frac{3n + 1}{57 + 2n} = \frac{3}{2}$.
		\end{proposition}

		\begin{proof}
			We ask whether $\limn \frac{3n + 1}{2n + 57} = \frac{3}{2}$.
			Let $\epsilon$ be an arbitrary real number so that $\epsilon > 0$.
			Let $N$ be the particular natural number $\ceil{\frac{169}{4 \epsilon}}$.
			Let $n$ be an arbitrary natural number so that $n > N$.
			By algebra, we see that:
			\begin{align*}
				\abs{\frac{3n + 1}{2n + 57} - \frac{3}{2}}
				&= \abs{\frac{2 \cdot (3n + 1) - 3 \cdot (2n + 57)}{2 \cdot (2n + 57)}} \\
				&= \abs{\frac{6n + 2 - 6n - 171}{4n + 114}} \\
				&= \abs{\frac{-169}{4n + 114}} \\
				&= \frac{169}{4n + 114} \\
				&< \frac{169}{4n} \\
				&< \frac{169}{4N} \\
				&\leq \frac{169}{4 \cdot \frac{169}{4 \epsilon}} \\
				&= \epsilon
			\end{align*}
			Therefore, by the definition of limits, since for all $\epsilon > 0$ there exists $N \in \N$ so that $n > N$ implies $\abs{\frac{3n + 1}{2n + 57} - \frac{3}{2}} < \epsilon$, we see that $\limn \frac{3n + 1}{2n + 57} = \frac{3}{2}$.
		\end{proof}
	\end{problem}

	\begin{problem}[manual-num=63]
		\begin{proposition}
			Let $r_1, r_2, r_3, \dots$ be the sequence defined by $r_1 = 1$ and $r_{n + 1} = 4r_n + 7$ for all natural numbers $n$.
			Prove that $r_n = \frac{1}{3} (10 \cdot 4^{n-1} - 7)$ for all natural numbers $n$.
		\end{proposition}

		\begin{proof}
			Let $(r_n)_{n=1}^{\infty}$ be the sequence defined by $r_1 = 1$ and $r_{n + 1} = 4r_n + 7$ for all natural numbers $n$.
			We ask whether $r_n = \frac{1}{3} (10 \cdot 4^{n-1} - 7)$ for all natural numbers n.
			We will show this to be true through induction.

			Take the case where $n = 1$.
			By algebra, we see that:
			\begin{align*}
				\frac{1}{3} (10 \cdot 4^{(1)-1} - 7) &= \frac{1}{3} (10 \cdot 4^0 - 7) \\
				&= \frac{1}{3} (10 \cdot 1 - 7) \\
				&= \frac{1}{3} (10 - 7) \\
				&= \frac{1}{3} (3) \\
				&= 1 \\
				&= r_1
			\end{align*}
			By substitution, we see that $r_n = \frac{1}{3} (10 \cdot 4^{n-1} - 7)$ for $n = 1$.

			Let $k$ be an arbitrary natural number.
			Suppose $r_n = \frac{1}{3} (10 \cdot 4^{n-1} - 7)$ for $n = k$.
			Take the case where $n = k + 1$.
			By substitution, we see that $r_k = \frac{1}{3} (10 \cdot 4^{k-1} - 7)$.
			By algebra, we see that:
			\begin{align*}
				r_{k + 1} &= 4r_k + 7 \\
				&= 4 \cdot \frac{1}{3} (10 \cdot 4^{k-1} - 7) + 7 \\
				&= \frac{1}{3} \cdot (40 \cdot 4^{k - 1}) - \frac{1}{3} \cdot (28) + 7\\
				&= \frac{1}{3} \cdot (10 \cdot 4^k) - \frac{1}{3} \cdot (28) + \frac{1}{3} \cdot (21) \\
				&= \frac{1}{3} \cdot (10 \cdot 4^{(k - 1) + 1}) - \frac{1}{3} \cdot 7 \\
				&= \frac{1}{3} (10 \cdot 4^{(k - 1) + 1} - 7) \\
			\end{align*}
			By substitution, we see that $r_n = \frac{1}{3} (10 \cdot 4^{n-1} - 7)$ for $n = k + 1$.

			Therefore, by induction, we see that for the sequence  $(r_n)_{n=1}^{\infty}$ defined by $r_1 = 1$ and $r_{n + 1} = 4r_n + 7$, $r_n = \frac{1}{3} (10 \cdot 4^{n-1} - 7)$ for all natural numbers $n$.

		\end{proof}
	\end{problem}

	\begin{problem}[manual-num=69]
		\begin{proposition}
			Use the definition of a limit to prove $\limn \frac{n^2}{5 - n^2} = -1$.
		\end{proposition}

		\begin{proof}
			We ask whether $\limn \frac{n^2}{5 - n^2} = -1$.
			Let $\epsilon$ be an arbitrary real number so that $\epsilon > 0$.
			Let $N$ be the particular natural number $\ceil{({\frac{5}{\epsilon} + 5})^{\frac{1}{2}}}$.
			Let $n$ be an arbitrary natural number so that $n > N$.
			By substitution, we see that $N \geq \ceil{({\frac{5}{\epsilon} + 5)^{\frac{1}{2}}}} \geq \ceil{5^\frac{1}{2}} = 3$.
			By substitution, we see that $n > N > 3$.
			By substitution, we see that $n^2 > N^2 > 3^2 = 9 > 5$.
			By algebra, we see that:
			\begin{align*}
				\abs{\frac{n^2}{5 - n^2} - (-1)}
				&= \abs{\frac{n^2}{5 - n^2} + 1} \\
				&= \abs{\frac{n^2}{5 - n^2} + \frac{5 - n^2}{5 - n^2}} \\
				&= \abs{\frac{n^2 + (5 - n^2)}{5 - n^2}} \\
				&= \abs{\frac{5}{5 - n^2}} \\
				&= \frac{5}{n^2 - 5} \quad \text{(since $n^2 > 5$)} \\
				&< \frac{5}{N^2 - 5} \quad \text{(since $N^2 > 5$)} \\
				&\leq \frac{5}{((\frac{5}{\epsilon} + 5)^{\frac{1}{2}})^2 - 5} \\
				&= \frac{5}{(\frac{5}{\epsilon} + 5) - 5} \\
				&= \frac{5}{\frac{5}{\epsilon}} \\
				&= \epsilon
			\end{align*}
			Therefore, by the definition of limits, since for all $\epsilon > 0$ there exists $N \in \N$ so that $n > N$ implies $\abs{\frac{n^2}{5 - n^2} - (-1)} < \epsilon$, we see that $\limn \frac{n^2}{5 - n^2} = -1$.
		\end{proof}
	\end{problem}

	\begin{problem}[manual-num=74]
		\begin{proposition}
			Use strong induction to prove that every natural number can be written as a sum of distinct powers of 2.
		\end{proposition}

		\begin{proof}
			Let $P(n)$ be the statement that for a natural number $n$, there exists a finitely supported sequence $(S^n_i)_{i = 0}^\infty$ of elements of $\set{0, 1}$ so that $n = \sum_{i = 0}^\infty 2^i \cdot S^n_i$.
			We ask whether $P(n)$ for all natural numbers $n$.
			We will show this to be true through strong induction.

			Take the case where $n = 1$.
			Let $(S^1_i)_{i \in \N}$ be the finitely supported sequence defined by $S^1_0 = 1$ and $S^1_i = 0$ for all $i \in \N^+$.
			We see that $\sum_{i = 0}^\infty 2^i \cdot S^1_i = 2^0 \cdot 1 = 1$ and $P(n)$ for $n = 1$.

			Let $k$ be an arbitrary natural number.
			Suppose $P(n)$ for all natural numbers $n \leq k$.
			Take the case where $n = k + 1$.

			Take the case where $k + 1$ is even.
			By the definition of parity, we see that $\frac{1}{2}(k + 1) \in \N$.
			Since $\frac{1}{2}(k + 1) \leq k$, we see that there exists a finitely supported sequence $(S^{\frac{1}{2}(k + 1)}_i)_{i = 0}^\infty$ of elements of $\set{0, 1}$ so that $\frac{1}{2}(k + 1) = \sum_{i = 0}^\infty 2^i \cdot S^{\frac{1}{2}(k + 1)}_i$.
			Let $(S^{k + 1}_i)_{i = 0}^\infty$ be the finitely supported sequence defined by $S^{k + 1}_0 = 0$ and $S^{k + 1}_i = S^{\frac{1}{2}(k + 1)}_{i - 1}$ for all $i \in \N^+$.
			By algebra, we see that:
			\begin{align*}
				\frac{1}{2}(k + 1) &= \sum_{i = 0}^\infty 2^i \cdot S^{\frac{1}{2}(k + 1)}_i \\
				k + 1 &= 2 \cdot \sum_{i = 0}^\infty 2^i \cdot S^{\frac{1}{2}(k + 1)}_i \\
				&= \sum_{i = 0}^\infty 2^{i + 1} \cdot S^{\frac{1}{2}(k + 1)}_i \\
				&= \sum_{i = 1}^\infty 2^i \cdot S^{\frac{1}{2}(k + 1)}_{i - 1}\\
				&= S^{k + 1}_0 + \sum_{i = 1}^\infty 2^i \cdot S^{\frac{1}{2}(k + 1)}_{i - 1} \quad \text{(since $S^{k + 1}_0 = 0$)} \\
				&= \sum_{i = 0}^\infty 2^i \cdot S^{k + 1}_i \quad \text{(since $S^{k + 1}_i = S^{\frac{1}{2}(k + 1)}_{i - 1}$ for all $i \in \N^+$)}
			\end{align*}
			By substitution, we see that $P(k + 1)$ if $k + 1$ is even.

			Take the case where $k + 1$ is odd.
			Since $P(k)$, we see that there exists a finitely supported sequence $(S^k_i)_{i = 0}^\infty$ of elements of $\set{0, 1}$ so that $k = \sum_{i = 0}^\infty 2^i \cdot S^k_i$.
			By the definition of parity, since $k + 1$ is odd, we see that $k$ is even.
			Since $\sum_{i = 0}^\infty 2^i \cdot S^k_i$ is even, $2^i$ is even for all $i > 0$, $2^0$ is odd, and the sum of an even and an odd is odd, we see that $2^0 \cdot S^k_0 = 0$.
			Let $(S^{k + 1}_i)_{i = 0}^\infty$ be the finitely supported sequence defined by $S^{k + 1}_0 = 1$ and $S^{k + 1}_i = S^k_i$ for all $i \in \N^+$.
			By algebra, we see that:
			\begin{align*}
				k + 1 &= 1 + \sum_{i = 0}^\infty 2^i \cdot S^k_i \\
				&= S^{k + 1}_0 + \sum_{i = 0}^\infty 2^i \cdot S^k_i \quad \text{(since $S^{k + 1}_0 = 1$)} \\
				&= S^{k + 1}_0 + \sum_{i = 1}^\infty 2^i \cdot S^k_i \quad \text{(since $S^k_0 = 0$)} \\
				&= \sum_{i = 0}^\infty 2^i \cdot S^{k + 1}_i
			\end{align*}
			By substitution, we see that $P(k + 1)$ if $k + 1$ is odd.

			Since $k + 1$ must be even or odd, we see that $P(k + 1)$ in either case.
			Therefore, by induction, we see that $P(n)$ for all natural numbers $n$.
		\end{proof}
	\end{problem}
